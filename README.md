public class CoracaoNordestino {
    
    public static void enfrentarDesafio(String intensidadeDaDor) {
        if (intensidadeDaDor.equals("imensa") || intensidadeDaDor.equals("forte")) {
            Vida.acolherETransformar(intensidadeDaDor);
        } else {
            System.out.println("O coração do sertanejo sempre encontra força para superar.");

    static class Vida {

    }
}
